Что такое Seismic?

Шифрование на EVM

Seismic вводит шифрование непосредственно в Ethereum Virtual Machine (EVM), позволяя защищать конфиденциальные данные в смарт-контрактах.

Специальные типы данных

Используйте типы suint, saddress и sbool для автоматического шифрования чувствительной информации в контрактах.

Прозрачное исполнение

Несмотря на шифрование данных, выполнение смарт-контрактов остается проверяемым, сохраняя преимущества децентрализации.

Как работает Seismic?

Шифрование данных

1. Шифрование данных транзакции

Когда вы используете типы данных с префиксом 's' (например, saddress или suint), Seismic автоматически шифрует эти значения перед отправкой транзакции в сеть.

function payRent(saddress landlord, suint256 amount) external { // Адрес landlord и сумма amount зашифрованы! }
Trusted Execution Environment

2. Исполнение в Trusted Execution Environment

Seismic использует технологию TEE (Trusted Execution Environment) для безопасного выполнения зашифрованных смарт-контрактов без раскрытия данных.

TEE обеспечивает изолированную среду, где код выполняется безопасно, а данные защищены от внешнего доступа.
Верификация без раскрытия

3. Верификация без раскрытия данных

Транзакции верифицируются в сети без раскрытия зашифрованных данных, обеспечивая как приватность, так и целостность блокчейна.

Как это работает?

                                
Seismic использует Homomorphic Encryption и Zero-Knowledge Proofs для защиты данных:

                                
    
                                    
  1. Данные шифруются на стороне клиента с использованием FHE (Fully Homomorphic Encryption)
    2. 
                                    
  2. Зашифрованные данные отправляются в блокчейн и обрабатываются без расшифровки
    3. 
                                    
  3. Trusted Execution Environment (TEE) обрабатывает данные, сохраняя конфиденциальность
    4. 
                                    
  4. ZK-доказательства подтверждают корректность операций без раскрытия данных

Пример контракта для конфиденциальных платежей

                    
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

contract PrivatePayments {
    // Сопоставление адреса с зашифрованным балансом
    mapping(address => suint256) private balances;
    
    // Событие о пополнении средств
    event Deposit(address indexed account);
    
    // Событие о переводе средств
    event Transfer(address indexed from);
    
    // Пополнение баланса
    function deposit(suint256 amount) external payable {
        // Увеличиваем зашифрованный баланс
        balances[msg.sender] += amount;
        
        emit Deposit(msg.sender);
    }
    
    // Приватный перевод средств
    function transfer(saddress to, suint256 amount) external {
        // Проверяем, что у отправителя достаточно средств
        require(balances[msg.sender] >= amount, "Insufficient balance");
        
        // Уменьшаем баланс отправителя
        balances[msg.sender] -= amount;
        
        // Увеличиваем баланс получателя
        // Зашифрованный адрес 'to' расшифровывается внутри TEE
        address recipient = to; // неявное преобразование
        balances[recipient] += amount;
        
        emit Transfer(msg.sender);
    }
}
                    

                    
Этот контракт позволяет:

                    
    
                        
  • Хранить зашифрованные балансы пользователей
    • 
                        
  • Выполнять переводы без раскрытия суммы или получателя
    • 
                        
  • Проверять достаточность средств без их раскрытия

Приватное голосование на Seismic

                    
                

                

                    
Пример контракта для приватного голосования

                    
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

contract PrivateVoting {
    struct Proposal {
        string description;
        suint256 yesVotes;
        suint256 noVotes;
        uint256 endTime;
        bool finalized;
    }
    
    // Список предложений
    mapping(uint256 => Proposal) public proposals;
    
    // Отслеживание голосования
    mapping(uint256 => mapping(address => bool)) private hasVoted;
    
    // Событие создания предложения
    event ProposalCreated(uint256 indexed proposalId, string description, uint256 endTime);
    
    // Событие голосования
    event Voted(uint256 indexed proposalId, address voter);
    
    // Создание нового предложения
    function createProposal(string memory description, uint256 duration) external returns (uint256) {
        uint256 proposalId = uint256(keccak256(abi.encodePacked(description, block.timestamp)));
        
        proposals[proposalId] = Proposal({
            description: description,
            yesVotes: suint256(0),
            noVotes: suint256(0),
            endTime: block.timestamp + duration,
            finalized: false
        });
        
        emit ProposalCreated(proposalId, description, block.timestamp + duration);
        return proposalId;
    }
    
    // Голосование за предложение
    function vote(uint256 proposalId, sbool choice) external {
        Proposal storage proposal = proposals[proposalId];
        
        require(block.timestamp < proposal.endTime, "Voting period ended");
        require(!hasVoted[proposalId][msg.sender], "Already voted");
        
        // Зашифрованный выбор обрабатывается в TEE
        if (choice) {
            proposal.yesVotes += suint256(1);
        } else {
            proposal.noVotes += suint256(1);
        }
        
        hasVoted[proposalId][msg.sender] = true;
        
        emit Voted(proposalId, msg.sender);
    }
}
                    

                    
Преимущества приватного голосования на Seismic:

                    
    
                        
  • Голоса остаются зашифрованными, что предотвращает влияние на других участников
    • 
                        
  • Подсчет голосов происходит автоматически и безопасно внутри TEE
    • 
                        
  • Невозможно узнать, как проголосовал конкретный пользователь
    •

Скрытые ставки на аукционах

                    
                

                

                    
Пример контракта для аукциона со скрытыми ставками

                    
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

contract PrivateAuction {
    struct Auction {
        address seller;
        string itemDescription;
        uint256 endTime;
        bool finalized;
        address highestBidder;
        suint256 highestBid;
    }
    
    // Аукционы по ID
    mapping(uint256 => Auction) public auctions;
    
    // Ставки участников
    mapping(uint256 => mapping(address => suint256)) private bids;
    
    // События
    event AuctionCreated(uint256 indexed auctionId, string itemDescription, uint256 endTime);
    event BidPlaced(uint256 indexed auctionId, address bidder);
    event AuctionFinalized(uint256 indexed auctionId, address winner);
    
    // Создание аукциона
    function createAuction(string memory itemDescription, uint256 duration) external returns (uint256) {
        uint256 auctionId = uint256(keccak256(abi.encodePacked(itemDescription, block.timestamp)));
        
        auctions[auctionId] = Auction({
            seller: msg.sender,
            itemDescription: itemDescription,
            endTime: block.timestamp + duration,
            finalized: false,
            highestBidder: address(0),
            highestBid: suint256(0)
        });
        
        emit AuctionCreated(auctionId, itemDescription, block.timestamp + duration);
        return auctionId;
    }
    
    // Размещение ставки
    function makeBid(uint256 auctionId, suint256 bidAmount) external payable {
        Auction storage auction = auctions[auctionId];
        
        require(block.timestamp < auction.endTime, "Auction ended");
        require(!auction.finalized, "Auction already finalized");
        
        // Сохраняем ставку пользователя
        bids[auctionId][msg.sender] = bidAmount;
        
        // Обновляем наивысшую ставку если текущая выше
        if (bidAmount > auction.highestBid) {
            auction.highestBid = bidAmount;
            auction.highestBidder = msg.sender;
        }
        
        emit BidPlaced(auctionId, msg.sender);
    }
    
    // Завершение аукциона
    function finalizeAuction(uint256 auctionId) external {
        Auction storage auction = auctions[auctionId];
        
        require(msg.sender == auction.seller, "Only seller can finalize");
        require(block.timestamp >= auction.endTime, "Auction not ended yet");
        require(!auction.finalized, "Already finalized");
        
        auction.finalized = true;
        
        emit AuctionFinalized(auctionId, auction.highestBidder);
    }
}
                    

                    
Преимущества скрытых ставок:

                    
    
                        
  • Участники не видят ставки других участников до завершения аукциона
    • 
                        
  • Предотвращает последовательное повышение ставок
    • 
                        
  • Обеспечивает справедливую конкуренцию без знания минимальной суммы для победы
